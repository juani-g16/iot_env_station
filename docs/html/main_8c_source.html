187void task_show_data_oled(void *args)
188{
189 dht_data_t sensorData = {0};
190 u8g2_t u8g2;
191 // OLED Display Setup
192 u8g2 = init_oled_display();
193
194 while (true)
195 {
196 if (xQueueReceive(displayQueue, &sensorData, portMAX_DELAY))
197 {
198 struct tm timeinfo;
199 char temp_str[20], hum_str[20], date_str[17], time_str[12];
200 // Parse ISO8601 string back to time
201 strptime(sensorData.timestamp, "%Y-%m-%dT%H:%M:%SZ", &timeinfo);
202 // Format the date as "dd/mm/YYYY"
203 strftime(date_str, sizeof(date_str), "Date: %d/%m/%Y", &timeinfo);
204 // Format the time as "HH:mm"
205 strftime(time_str, sizeof(time_str), "Time: %H:%M", &timeinfo);
206 sprintf(temp_str, "Temp: %.2fC", sensorData.temperature);
207 sprintf(hum_str, "Hum: %.2f %%", sensorData.humidity);
208 show_dht_data(u8g2, date_str, time_str, temp_str, hum_str);
209 }
210 else
211 {
212 ESP_LOGE(TAG, "Error receiving data or no data in buffer");
213 }
214 vTaskDelay(pdMS_TO_TICKS(100));
215 }
216}

234void task_send_data_mqtt(void *args)
235{
236 EventBits_t bits = xEventGroupWaitBits(s_wifi_event_group,
237 WIFI_CONNECTED_BIT,
238 pdFALSE,
239 pdTRUE,
240 pdMS_TO_TICKS(portMAX_DELAY));
241
242 if ((bits & WIFI_CONNECTED_BIT) == 1)
243 {
244 ESP_LOGI(TAG, "WiFi connected, starting MQTT task and SNTP");
245 setup_mqtt();
246 setup_sntp();
247 }
248
249 dht_data_t sensorData = {0};
250 while (true)
251 {
252 if ((xQueueReceive(mqttQueue, &sensorData, portMAX_DELAY)) && MQTT_CONNECTED)
253 {
254 // Convert to JSON string
255 char *json_str = create_json_payload(&sensorData);
256 esp_mqtt_client_publish(client, "/home/office/dht", json_str, 0, 0, 0);
257 free(json_str);
258 }
259 else
260 {
261 ESP_LOGE(TAG, "Error receiving data or no data in buffer");
262 }
263 vTaskDelay(pdMS_TO_TICKS(100));
264 }
265}

318void measure_temp_hum(TimerHandle_t timer)
319{
320 esp_err_t res;
321 dht_data_t dhtData;
322 res = dht_read_float_data(DHT_SENSOR_TYPE, CONFIG_ESP_TEMP_SENSOR_GPIO, &dhtData.humidity, &dhtData.temperature);
323 get_current_date_time(dhtData.timestamp);
324 if (res == ESP_OK)

333 }
